solution : 

from operator import itemgetter

if __name__ == '__main__':
    table = []
   
    
    for _ in range(int(input())):
        name = input()
        score = float(input())
        table.append((name,score))
        
    sorted_table = sorted(table, key=itemgetter(1,0))
    lowest_score = sorted_table[0][1]
    
    for _ , j in sorted_table:
        if j > lowest_score :
            second_lowest_score = j
            break
             
    for name,score in sorted_table:
        if score == second_lowest_score :
            print(name)
