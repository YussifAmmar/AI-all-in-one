solution : 
if __name__ == '__main__':
    n = int(input())
    student_marks = {}
    for _ in range(n):
        name, *line = input().split()
        scores = list(map(float, line))
        student_marks[name] = scores
    query_name = input()

    avg_score={}
    for name , scores in student_marks.items():
        avg_score[name]=sum(student_marks[name])/3
        
    print(f"{avg_score[query_name]:.2f}")
